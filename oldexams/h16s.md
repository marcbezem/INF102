public boolean reach(Integer s, Integer t, boolean[] marked){
  if (s==t) return true; // returns as early as possible
  marked[s] = true;
  for (Integer v : adj[s])
    if (!marked[v] &&  reach(v,t,marked)) return true;
  return false;
}
